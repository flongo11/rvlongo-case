    # CREATE SHA256 HASH VERIFICATION FOR YOUR MASTER PORTFOLIO
    Write-Host "`n=== CREATING SHA256 HASH VERIFICATION ===" -ForegroundColor Cyan

    # Function to generate SHA256 hashes for evidence files
    function Get-EvidenceHashes {
        $evidenceFiles = Get-ChildItem "D:\CASE_MASTER_2025\04_EVIDENCE_DOCUMENTS" -File
        
        $hashes = foreach ($file in $evidenceFiles) {
            $hash = Get-FileHash $file.FullName -Algorithm SHA256
            [PSCustomObject]@{
                File = $file.Name
                SHA256 = $hash.Hash
                Size_KB = [math]::Round($file.Length/1KB, 2)
                LastModified = $file.LastWriteTime
            }
        }
        
        return $hashes
    }

    # Generate and save hash verification
    $evidenceHashes = Get-EvidenceHashes
