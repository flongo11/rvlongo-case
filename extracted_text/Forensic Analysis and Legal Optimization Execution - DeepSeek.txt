                    }
                }
            }
        }
        catch {

            TemporalImpossibility = $TemporalImpossibility

    }

    function Test-Exclusion {
        param([System.IO.FileInfo]$File)
        
        $FileName = $File.Name

    # =============================================
    # FILE CLASSIFICATION ENGINE
    # =============================================
    function Classify-File {
        param([System.IO.FileInfo]$File)
        
        $FileName = $File.Name
        $FilePath = $File.FullName
        $CreationDate = $File.CreationTime
        $ModifiedDate = $File.LastWriteTime
        
        # Initialize classification
        $Classification = @{
            FileName = $FileName
            FilePath = $FilePath
            CreationDate = $CreationDate
            ModifiedDate = $ModifiedDate
            FileSize = "{0:N1} MB" -f ($File.Length / 1MB)
            IsPoliceDisclosure = $false
            Contains2022 = $false
            PoliceEvidence = @()
            FabricationContext = @()
            PDFMetadata = @{}
            DocumentType = "Unknown"
            RiskLevel = "Low"
        }
        
        # TEST 1: Is this police/Crown disclosure?
        $PoliceEvidence = Test-PoliceDisclosure -File $File
        if ($PoliceEvidence.IsPoliceDocument) {
            $Classification.IsPoliceDisclosure = $true
            $Classification.PoliceEvidence = $PoliceEvidence.Evidence
            $Classification.DocumentType = $PoliceEvidence.DocumentType
            
            # TEST 2: Does it contain 2022 fabrication?
            $FabricationAnalysis = Test-2022FabricationInContext -FilePath $File.FullName
            if ($FabricationAnalysis.Contains2022) {
                $Classification.Contains2022 = $true
                $Classification.FabricationContext = $FabricationAnalysis.Context
                $Classification.RiskLevel = "High"
                
                # TEST 3: Extract PDF metadata for deeper analysis
                if ($File.Extension -eq ".pdf") {
                    $Classification.PDFMetadata = Extract-PDFMetadata -FilePath $File.FullName
                }
            }
        }
        
        return $Classification
    }

    # =============================================
    # POLICE DISCLOSURE IDENTIFICATION
    # =============================================
    function Test-PoliceDisclosure {
        param([System.IO.FileInfo]$File)
        
        $FileName = $File.Name.ToLower()
        $FilePath = $File.FullName.ToLower()
        $IsPoliceDocument = $false
        $Evidence = @()
        $DocumentType = "Unknown"
        
        # Test file name and path for police keywords
        foreach ($Keyword in $PoliceKeywords) {
            $LowerKeyword = $Keyword.ToLower()
            if ($FileName -match $LowerKeyword -or $FilePath -match $LowerKeyword) {
                $IsPoliceDocument = $true
                $Evidence += "Keyword in path/name: $Keyword"
                
                # Classify document type
                switch -Wildcard ($Keyword) {
                    "Police" { $DocumentType = "Police Report" }
                    "Officer" { $DocumentType = "Officer Statement" }
                    "Crown" { $DocumentType = "Crown Brief" }
                    "Statement" { $DocumentType = "Witness Statement" }
                    "Will Say" { $DocumentType = "Will Say Statement" }
                    "Booking" { $DocumentType = "Booking Record" }
                    "Occurrence" { $DocumentType = "Occurrence Report" }
                }
            }
        }
        
        # Test content for police terminology (if not already identified)
        if (-not $IsPoliceDocument) {
            try {
                $Content = Get-Content -Path $File.FullName -Raw -ErrorAction SilentlyContinue
                if ($Content -ne $null) {
                    foreach ($Keyword in $PoliceKeywords) {
                        if ($Content -match $Keyword) {
                            $IsPoliceDocument = $true
                            $Evidence += "Keyword in content: $Keyword"
                            break
                        }
                    }
                }
            }
            catch {
                # Content unreadable, rely on metadata only
            }
        }
        
        return @{
            IsPoliceDocument = $IsPoliceDocument
            Evidence = $Evidence
            DocumentType = $DocumentType
        }
    }

    # =============================================
    # 2022 FABRICATION IN CONTEXT DETECTION
    # =============================================
    function Test-2022FabricationInContext {
        param([string]$FilePath)
        
        $Contains2022 = $false
        $Context = @()
        
        try {
            $Content = Get-Content -Path $FilePath -Raw -ErrorAction SilentlyContinue
            
            if ($Content -ne $null) {
                # Look for 2022 in context of arrest/incident patterns
                foreach ($Pattern in $FabricationPatterns) {
                    $Matches = [regex]::Matches($Content, $Pattern, [System.Text.RegularExpressions.RegexOptions]::IgnoreCase)
                    
                    foreach ($Match in $Matches) {
                        $Contains2022 = $true
                        
                        # Extract extended context
                        $StartPos = [Math]::Max(0, $Match.Index - 100)
                        $Length = [Math]::Min(200, $Content.Length - $StartPos)
                        $TextContext = $Content.Substring($StartPos, $Length).Replace("`n", " ").Replace("`r", " ")
                        $Context += "[$Pattern] -> ...$TextContext..."
                    }
                }
                
                # Also look for standalone 2022 near date references
                $DateMatches = [regex]::Matches($Content, "\b(2022)\b", [System.Text.RegularExpressions.RegexOptions]::IgnoreCase)
                foreach ($Match in $DateMatches) {
                    # Check if near date-related words
                    $StartPos = [Math]::Max(0, $Match.Index - 50)
                    $Length = [Math]::Min(100, $Content.Length - $StartPos)
                    $TextContext = $Content.Substring($StartPos, $Length)
                    
                    if ($TextContext -match "(date|arrest|incident|occurrence|booking|may|16)") {
                        $Contains2022 = $true
                        $Context += "[Standalone 2022] -> ...$TextContext..."
                    }
                }
            }
        }
        catch {
            $Context += "CONTENT_READ_ERROR: $($_.Exception.Message)"
        }
        
        return @{
            Contains2022 = $Contains2022
            Context = $Context
        }
    }

    # =============================================
    # PDF METADATA EXTRACTION
    # =============================================
    function Extract-PDFMetadata {
        param([string]$FilePath)
        
        $Metadata = @{}
        
        try {
            # Method 1: Use Shell Application for basic metadata
            $Shell = New-Object -ComObject Shell.Application
            $Folder = $Shell.NameSpace((Get-Item $FilePath).DirectoryName)
            $File = $Folder.ParseName((Get-Item $FilePath).Name)
            
            # Common metadata properties
            $PropertyNames = @{
                4 = "Author"; 5 = "Title"; 6 = "Subject"; 7 = "Category"; 8 = "Pages";
                9 = "Comments"; 10 = "Copyright"; 11 = "Artist"; 12 = "AlbumTitle";
                13 = "Year"; 14 = "TrackNumber"; 15 = "Genre"; 16 = "Duration";
                17 = "BitRate"; 18 = "Protected"; 19 = "CameraModel"; 20 = "DateTaken";
                21 = "Dimensions"; 22 = "CameraMaker"; 23 = "Company"; 24 = "Description";
                25 = "FileVersion"; 26 = "ProductName"; 27 = "ProductVersion"
            }
            
            for ($i = 0; $i -le 300; $i++) {
                $PropertyValue = $Folder.GetDetailsOf($File, $i)
                if ($PropertyValue -and $PropertyNames.ContainsKey($i)) {
                    $PropertyName = $PropertyNames[$i]
                    $Metadata[$PropertyName] = $PropertyValue
                }
            }
        }
        catch {
            $Metadata["ExtractionError"] = $_.Exception.Message
        }
        
        return $Metadata
    }

    function Get-AllDisclosureFilesWithExclusion {
        $AllFiles = @()
        
        foreach ($Drive in $SearchDrives) {
            if (Test-Path $Drive) {
                Write-Host "`n   Scanning $Drive for official documents..." -ForegroundColor White
                $Files = Get-ChildItem -Path $Drive -Recurse -Include $OfficialExtensions -ErrorAction SilentlyContinue
                
                foreach ($File in $Files) {
                    # Apply strict exclusion filtering
                    if (-not (Test-StrictExclusion -File $File)) {
                        $AllFiles += $File
                    }
                }
            }
        }
        
        return $AllFiles
    }

        param([System.IO.FileInfo]$File)

        $FilePath = $File.FullName
        $CreationDate = $File.CreationTime
        $ModifiedDate = $File.LastWriteTime

            DocumentType = $DocumentType

    # 2022 FABRICATION CONTEXT DETECTION

    function Test-2022FabricationContext {
        param([string]$FilePath)
        
        $Contains2022 = $false
        $Context = @()
        
        try {
            $Content = Get-Content -Path $FilePath -Raw -ErrorAction SilentlyContinue
            
            if ($Content -ne $null) {
                # Look for 2022 in specific fabrication patterns
                foreach ($Pattern in $FabricationPatterns) {
                    $Matches = [regex]::Matches($Content, $Pattern, [System.Text.RegularExpressions.RegexOptions]::IgnoreCase)
                    
                    foreach ($Match in $Matches) {
                        $Contains2022 = $true
                        
                        # Extract extended context (200 chars)
                        $StartPos = [Math]::Max(0, $Match.Index - 100)
                        $Length = [Math]::Min(200, $Content.Length - $StartPos)
                        $TextContext = $Content.Substring($StartPos, $Length).Replace("`n", " ").Replace("`r", " ").Trim()
                        $Context += "[$Pattern] -> ...$TextContext..."
                    }
                }
                
                # Additional check: standalone 2022 near date references
                if (-not $Contains2022) {
                    $DateMatches = [regex]::Matches($Content, "\b(2022)\b", [System.Text.RegularExpressions.RegexOptions]::IgnoreCase)
                    foreach ($Match in $DateMatches) {
                        # Check if near date-related words within 50 characters
                        $StartPos = [Math]::Max(0, $Match.Index - 50)
                        $Length = [Math]::Min(100, $Content.Length - $StartPos)
                        $TextContext = $Content.Substring($StartPos, $Length)
                        
                        if ($TextContext -match "(date|arrest|incident|occurrence|booking|may|16|charge|offense)") {
                            $Contains2022 = $true
                            $Context += "[Standalone 2022 near date reference] -> ...$TextContext..."
                        }
                    }
                }
            }
        }
        catch {
            $Context += "CONTENT_READ_ERROR: $($_.Exception.Message)"
        }
        
        return @{
            Contains2022 = $Contains2022
            Context = $Context
        }
    }

    function Extract-PDFMetadataAdvanced {
        param([string]$FilePath)
        
        $Metadata = @{}
        
        try {
            # Method 1: Use Shell.Application COM object
            $Shell = New-Object -ComObject Shell.Application
            $Folder = $Shell.NameSpace((Get-Item $FilePath).DirectoryName)
            $File = $Folder.ParseName((Get-Item $FilePath).Name)
            
            # Extended metadata properties
            $PropertyNames = @{
                0 = "Name"; 1 = "Size"; 2 = "Item type"; 3 = "Date modified";
                4 = "Date created"; 5 = "Date accessed"; 6 = "Attributes";
                7 = "Offline status"; 8 = "Availability"; 9 = "Perceived type";
                10 = "Owner"; 11 = "Kind"; 12 = "Date taken"; 13 = "Contributors";
                14 = "Subject"; 15 = "Title"; 16 = "Rating"; 17 = "Tags";
                18 = "Author"; 19 = "Last saved by"; 20 = "Revision number";
                21 = "Program name"; 22 = "Date last saved"; 23 = "Total editing time";
                24 = "Pages"; 25 = "Words"; 26 = "Characters"; 27 = "Application";
                28 = "Security"; 29 = "Category"; 30 = "Format"; 31 = "Status"
            }
            
            for ($i = 0; $i -le 50; $i++) {
                try {
                    $PropertyValue = $Folder.GetDetailsOf($File, $i)
                    if ($PropertyValue -and $PropertyNames.ContainsKey($i)) {
                        $PropertyName = $PropertyNames[$i]
                        $Metadata[$PropertyName] = $PropertyValue.Trim()
                    }
                }
                catch {
                    # Skip inaccessible properties
                }
            }
            
            # Look for police/Crown identifiers in metadata
            foreach ($key in $Metadata.Keys) {
                $value = $Metadata[$key]
                if ($value -match "(windsor|police|crown|attorney|prosecutor|officer)") {
                    $Metadata["LawEnforcementIdentifier"] = "$key: $value"
                }
            }
        }
        catch {
            $Metadata["ExtractionError"] = $_.Exception.Message
        }
        
        return $Metadata
    }
