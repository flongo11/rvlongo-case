    # FORENSIC EMAIL ANALYSIS SCRIPT
    # Comprehensive Legal Evidence Collection

    param(
        [string]$ExportPath = "C:\ForensicEvidence\",
        [string]$GmailExportPath = "C:\GmailExports\",
        [switch]$GenerateHashes = $true,
        [switch]$AnalyzeEML = $true,
        [switch]$MonitorChanges = $true
    )

    # Create evidence directory structure
    $Directories = @(
        "EmailEvidence",
        "LegalDocuments", 
        "ForensicReports",
        "HashValues",
        "CourtReady",
        "TamperEvidence"
    )

    foreach ($dir in $Directories) {
        New-Item -Path (Join-Path $ExportPath $dir) -ItemType Directory -Force | Out-Null
    }

    function Get-ForensicSystemInfo {
        # System baseline for evidence integrity
        $SystemInfo = @{
            ComputerName = $env:COMPUTERNAME
            UserName = $env:USERNAME
            AnalysisDate = Get-Date
            OSVersion = (Get-WmiObject Win32_OperatingSystem).Caption
            TimeZone = (Get-TimeZone).DisplayName
        }
        return $SystemInfo
    }

    function New-ForensicHashManifest {
        param([string]$Path)
        
        $HashResults = @()
        Get-ChildItem -Path $Path -Recurse -File | ForEach-Object {
            $Hash = Get-FileHash -Path $_.FullName -Algorithm SHA256
            $HashResults += [PSCustomObject]@{
                FileName = $_.Name
                FilePath = $_.FullName
                SHA256 = $_.Hash
                FileSize = $_.Length
                LastModified = $_.LastWriteTime
                Created = $_.CreationTime
            }
        }
        
        $HashResults | Export-Csv -Path (Join-Path $ExportPath "HashValues\FileHashManifest.csv") -NoTypeInformation
        return $HashResults
    }

    function Analyze-EMLFiles {
        param([string]$EMLPath)
        
        $EMLAnalysis = @()
        Get-ChildItem -Path $EMLPath -Filter "*.eml" -Recurse | ForEach-Object {
            try {
                $Content = Get-Content -Path $_.FullName -Raw
                $Headers = [ordered]@{}
                
                # Extract email headers
                $HeaderSection = ($Content -split "`r`n`r`n" | Select-Object -First 1) -split "`r`n"
                foreach ($Line in $HeaderSection) {
                    if ($Line -match "^\s") { continue } # Skip continuation lines
                    if ($Line -match "^([^:]+):\s*(.+)") {
                        $Headers[$Matches[1]] = $Matches[2]
                    }
                }
                
                $EMLAnalysis += [PSCustomObject]@{
                    FileName = $_.Name
                    FilePath = $_.FullName
                    From = $Headers['From']
                    To = $Headers['To']
                    Subject = $Headers['Subject']
                    Date = $Headers['Date']
                    MessageID = $Headers['Message-ID']
                    Received = $Headers['Received']
                    ContentLength = $Content.Length
                    HasAttachments = ($Content -match "Content-Disposition: attachment")
                    AnalysisDate = Get-Date
                }
            }
            catch {
                Write-Warning "Error analyzing EML file: $($_.Exception.Message)"
            }
        }
        
        $EMLAnalysis | Export-Csv -Path (Join-Path $ExportPath "ForensicReports\EMLAnalysis.csv") -NoTypeInformation
        return $EMLAnalysis
    }

    function Monitor-FileChanges {
        param([string]$WatchPath)
        
        $Watcher = New-Object System.IO.FileSystemWatcher
        $Watcher.Path = $WatchPath
        $Watcher.IncludeSubdirectories = $true
        $Watcher.EnableRaisingEvents = $true
        
        $Action = {
            $path = $Event.SourceEventArgs.FullPath
            $changeType = $Event.SourceEventArgs.ChangeType
            $timestamp = Get-Date
            
            "$timestamp - $changeType - $path" | Out-File (Join-Path $ExportPath "TamperEvidence\FileMonitor.log") -Append
        }
        
        Register-ObjectEvent -InputObject $Watcher -EventName "Created" -Action $Action
        Register-ObjectEvent -InputObject $Watcher -EventName "Changed" -Action $Action
        Register-ObjectEvent -InputObject $Watcher -EventName "Deleted" -Action $Action
        Register-ObjectEvent -InputObject $Watcher -EventName "Renamed" -Action $Action
        
        return $Watcher
    }

    function New-EvidencePortfolio {
        # Create master evidence portfolio
        $Portfolio = @{
            SystemInfo = Get-ForensicSystemInfo
            AnalysisTimestamp = Get-Date
            EvidenceSummary = @{
                TotalFiles = 0
                TotalEmails = 0
                TamperEvents = 0
                LegalDocuments = 0
            }
            ChainOfCustody = @()
        }
        
        return $Portfolio
    }

    # MAIN EXECUTION
    Write-Host "Starting Forensic Analysis..." -ForegroundColor Green

    # 1. System Baseline
    Write-Host "1. Creating system baseline..." -ForegroundColor Yellow
    $SystemInfo = Get-ForensicSystemInfo
    $SystemInfo | ConvertTo-Json | Out-File (Join-Path $ExportPath "ForensicReports\SystemBaseline.json")

    # 2. Generate Hash Manifest
    if ($GenerateHashes) {
        Write-Host "2. Generating SHA256 hashes..." -ForegroundColor Yellow
        $Hashes = New-ForensicHashManifest -Path $GmailExportPath
        Write-Host "   Generated $($Hashes.Count) hash entries" -ForegroundColor Green
    }

    # 3. Analyze EML Files
    if ($AnalyzeEML -and (Test-Path $GmailExportPath)) {
        Write-Host "3. Analyzing EML files..." -ForegroundColor Yellow
        $EMLAnalysis = Analyze-EMLFiles -EMLPath $GmailExportPath
        Write-Host "   Analyzed $($EMLAnalysis.Count) EML files" -ForegroundColor Green
    }

    # 4. Start File Monitoring
    if ($MonitorChanges) {
        Write-Host "4. Starting file change monitoring..." -ForegroundColor Yellow
        $Watcher = Monitor-FileChanges -WatchPath $GmailExportPath
        Write-Host "   Monitoring active on: $GmailExportPath" -ForegroundColor Green
    }

    # 5. Create Evidence Portfolio
    Write-Host "5. Creating master evidence portfolio..." -ForegroundColor Yellow
    $Portfolio = New-EvidencePortfolio
    $Portfolio | ConvertTo-Json -Depth 5 | Out-File (Join-Path $ExportPath "CourtReady\MasterEvidencePortfolio.json")

    # Generate Summary Report
    $Summary = @"
    FORENSIC ANALYSIS COMPLETE
    ==========================
    Analysis Date: $(Get-Date)
    System: $($SystemInfo.ComputerName)
    User: $($SystemInfo.UserName)

    EVIDENCE COLLECTED:
    - System Baseline: Complete
    - File Hashes: $($Hashes.Count) files
    - EML Analysis: $($EMLAnalysis.Count) emails
    - Monitoring: $(if($MonitorChanges){'Active'}else{'Inactive'})

    Output Location: $ExportPath
    "@

    $Summary | Out-File (Join-Path $ExportPath "ForensicReports\AnalysisSummary.txt")
    Write-Host $Summary -ForegroundColor Cyan

    Write-Host "`nForensic analysis complete. Evidence ready for legal proceedings." -ForegroundColor Green
