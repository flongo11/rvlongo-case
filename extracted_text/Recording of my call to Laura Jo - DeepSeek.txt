    # Function to decode base64 and other encodings
    function Decode-EmailContent {
        param([string]$Content, [string]$FilePath)
        
        $results = @()
        
        # Try base64 decoding
        try {
            $base64Pattern = '[A-Za-z0-9+/]{20,}={0,2}'
            if ($Content -match $base64Pattern) {
                $base64Matches = [regex]::Matches($Content, $base64Pattern)
                foreach ($match in $base64Matches) {
                    try {
                        $decoded = [System.Text.Encoding]::UTF8.GetString([System.Convert]::FromBase64String($match.Value))
                        if ($decoded -match '[a-zA-Z]{3,}') { # Basic English text check
                            $results += "BASE64 DECODED: $decoded"
                        }
                    } catch {
                        # Not valid base64, continue
                    }
                }
            }
        } catch {
            # Base64 decoding failed
        }
        
        # Check for URL encoding
        if ($Content -match '%[0-9A-Fa-f]{2}') {
            try {
                $decoded = [System.Web.HttpUtility]::UrlDecode($Content)
                $results += "URL DECODED: $decoded"
            } catch {
                # URL decode failed
            }
        }
        
        # Check for MIME encoded words (common in emails)
        if ($Content -match '=\?[^\?]+\?[BQ]\?[^\?]+\?=') {
            $results += "MIME ENCODING DETECTED - Requires specialized email parser"
        }
        
        # Check for hex encoding
        if ($Content -match '[0-9A-Fa-f]{8,}') {
            try {
                $hexString = $Content -replace '[^0-9A-Fa-f]', ''
                if ($hexString.Length -ge 8) {
                    $decoded = ""
                    for ($i = 0; $i -lt $hexString.Length; $i += 2) {
                        $decoded += [char][Convert]::ToByte($hexString.Substring($i, 2), 16)
                    }
                    if ($decoded -match '[a-zA-Z]{3,}') {
                        $results += "HEX DECODED: $decoded"
                    }
                }
            } catch {
                # Hex decode failed
            }
        }
        
        return $results
    }
